#include<iostream>
#include<vector>
using namespace std;

vector<int> F(int n) {
    vector<int> Fnumbers;

    if (n >= 1) {
        Fnumbers.push_back(0);
    }

    if (n >= 2) {
        Fnumbers.push_back(1);
    }

    while (Fnumbers.size() < n) {
        int nextF = Fnumbers[Fnumbers.size() - 1] + Fnumbers[Fnumbers.size() - 2];
        Fnumbers.push_back(nextF);
    }

    return Fnumbers;
}

bool Prime(int number) {
    if (number <= 1) {
        return false;
    }

    for (int i = 2; i * i <= number; ++i) {
        if (number % i == 0) {
            return false;
        }
    }

    return true;
}

vector<int> removePrime(const vector<int>& numbers) {
    vector<int> notPrime;

    for (int num : numbers) {
        if (num == 0 || num == 1 || !Prime(num)) {
            notPrime.push_back(num);
        }
    }

    return notPrime;
}

vector<int> removeEven(const vector<int>& numbers) {
    vector<int> notEven;

    for (int num : numbers) {
        if (num % 2 != 0) {
            notEven.push_back(num);
        }
    }

    return notEven;
}

vector<int> removeOdd(const vector<int>& numbers) {
    vector<int> notOdd;

    for (int num : numbers) {
        if (num % 2 != 1) {
            notOdd.push_back(num);
        }
    }

    return notOdd;
}

int main() {
    int n;

    cout << "要素数= ";
    cin >> n;

    vector<int> Fnumbers = F(n);

    cout << "フィボナッチ数列\n";

    for (int i = 0; i < Fnumbers.size(); ++i) {
        cout << Fnumbers[i] << " ";
    }

    vector<int> notPrime = removePrime(Fnumbers);

    cout << "\n素数を除いたフィボナッチ数列\n";

    for (int i = 0; i < notPrime.size(); ++i) {
        cout << notPrime[i] << " ";
    }

    vector<int> notEven = removeEven(Fnumbers);

    cout << "\n偶数を除いたフィボナッチ数列\n";

    for (int i = 0; i < notEven.size(); ++i) {
        cout << notEven[i] << " ";
    }

    vector<int> notOdd = removeOdd(Fnumbers);

    cout << "\n奇数を除いたフィボナッチ数列\n";

    for (int i = 0; i < notOdd.size(); ++i) {
        cout << notOdd[i] << " ";
    }

    return 0;
}
